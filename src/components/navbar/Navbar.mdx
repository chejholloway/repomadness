# Navbar

Navbar component displays a navigation bar with HomeLink and NavbarIconContainer components.

```jsx
const Navbar: React.FC = () => {
  return (
    <nav className="flex justify-around py-4 bg-white/80 backdrop-blur-md shadow-md w-full fixed top-0 left-0 right-0 z-10 mb-8">
      {/* Display NavbarIconContainer component */}
      <NavbarIconContainer />
      <div className="flex items-center space-x-5">
        {/* Display HomeLink component */}
        <HomeLink />
      </div>
    </nav>
  );
};

export default Navbar;
